## 반복자 패턴(Iterator Pattern)
동일한 형태의 데이터 항목을 여러개 가지고 있는 것을 Container / Aggregator 라고 한다.  
이 종류에는 여러가지가 있다. (Array, List, Tree, Graph, Table 등)  
집합에서 하나씩 데이터를 꺼내오는 방법은 종류마다 다를 것이다.  
예를 들어, 트리 구조는 깊이 우선, 너비 우선 탐색 등 여러 방법으로 순회할 수도 있다.  
이를 통일된 하나의 방법으로 데이터를 가져오는 패턴이 반복자 패턴이다.  
<br/>

### 구현
```java
// 집합체를 의미하는 인터페이스
public interface Aggregator {
    // iterator 객체를 만들어서 반환해주는 메소드
    Iterator iterator();
}

// Aggregator의 구현체
// Item을 보관하는 역할
// 배열을 순회하는 iterator를 생성하고 Item을 관리
public class Array implements Aggregator {
    private Item[] items;

    public Array(Item[] items) {
        this.items = items;
    }

    // 구성 데이터를 얻음
    public Item getItems(int index) {
        return items[index];
    }

    // 배열의 데이터 개수
    public int getCount() {
        return items.length;
    }

    @Override
    public Iterator iterator() {
        return new ArrayIterator(this);
    }
}
```
```java
public interface Iterator {
    // Aggregator의 다음 데이터를 얻을 수 있다면 true를 반환
    boolean next();
    // 구성 데이터를 하나 반환
    Object current();
}

public class ArrayIterator implements Iterator {
    private Array array;
    private int index;

    public ArrayIterator(Array array) {
        this.array = array;
        this.index = -1;
    }

    @Override
    public boolean next() {
        index++;
        return index < array.getCount();
    }

    @Override
    public Object current() {
        return array.getItems(index);
    }
}
```
```java
public class Item {
    private String name;
    private int cost;

    public Item(String name, int cost) {
        this.name = name;
        this.cost = cost;
    }

    @Override
    public String toString() {
        return "Item{" +
                "name='" + name + '\'' +
                ", cost=" + cost +
                '}';
    }
}

public class Main {
    public static void main(String[] args) {
        Item[] items = {
                new Item("CPU", 1000),
                new Item("Keyboard", 2000),
                new Item("Mouse", 3000),
                new Item("HDD", 50)
        };

        Array array = new Array(items);
        Iterator iterator = array.iterator();

        while (iterator.next()) {
            Item item = (Item) iterator.current();
            System.out.println(item);
        }
    }
}
```
